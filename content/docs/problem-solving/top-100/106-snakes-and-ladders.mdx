#include <iostream>
#include <vector>
#include <queue>
using namespace std;

struct QueueEntry {
    int v;
    int dist;
};

int getMinDiceThrows(vector<int>& board, int N) {
    vector<bool> visited(N, false);
    queue<QueueEntry> q;

    visited[0] = true;
    q.push({0, 0});

    while (!q.empty()) {
        QueueEntry qe = q.front();
        q.pop();

        int v = qe.v;

        if (v == N - 1)
            return qe.dist;

        for (int j = v + 1; j <= min(v + 6, N - 1); j++) {
            if (!visited[j]) {
                QueueEntry a;
                a.dist = qe.dist + 1;
                visited[j] = true;

                if (board[j] != -1)
                    a.v = board[j];
                else
                    a.v = j;

                q.push(a);
            }
        }
    }

    return -1;
}

int main() {
    int N = 30;
    vector<int> board(N, -1);

    // Ladders
    board[2] = 21;
    board[4] = 7;
    board[10] = 25;
    board[19] = 28;

    // Snakes
    board[26] = 0;
    board[20] = 8;
    board[16] = 3;
    board[18] = 6;

    cout << "Min Dice throws required: " << getMinDiceThrows(board, N) << endl;

    return 0;
}

from collections import deque

def get_min_dice_throws(board, N):
    visited = [False] * N
    queue = deque([(0, 0)])  # (position, distance)
    visited[0] = True

    while queue:
        v, dist = queue.popleft()

        if v == N - 1:
            return dist

        for j in range(v + 1, min(v + 7, N)):
            if not visited[j]:
                visited[j] = True

                if board[j] != -1:
                    next_pos = board[j]
                else:
                    next_pos = j

                queue.append((next_pos, dist + 1))

    return -1

function getMinDiceThrows(board, N) {
    const visited = new Array(N).fill(false);
    const queue = [[0, 0]]; // [position, distance]
    visited[0] = true;

    while (queue.length > 0) {
        const [v, dist] = queue.shift();

        if (v === N - 1) {
            return dist;
        }

        for (let j = v + 1; j < Math.min(v + 7, N); j++) {
            if (!visited[j]) {
                visited[j] = true;

                const nextPos = board[j] !== -1 ? board[j] : j;
                queue.push([nextPos, dist + 1]);
            }
        }
    }

    return -1;
}

// Example
const N = 30;
const board = new Array(N).fill(-1);

// Ladders
board[2] = 21;
board[4] = 7;
board[10] = 25;
board[19] = 28;

// Snakes
board[26] = 0;
board[20] = 8;
board[16] = 3;
board[18] = 6;

console.log("Min Dice throws required:", getMinDiceThrows(board, N));

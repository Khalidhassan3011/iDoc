#include <iostream>
#include <vector>
using namespace std;

bool searchWord(vector<vector<char>>& grid, string word, int row, int col, int index) {
    if (index == word.length())
        return true;

    int m = grid.size();
    int n = grid[0].size();

    if (row < 0 || row >= m || col < 0 || col >= n || grid[row][col] != word[index])
        return false;

    char temp = grid[row][col];
    grid[row][col] = '#';

    bool found = searchWord(grid, word, row + 1, col, index + 1) ||
                 searchWord(grid, word, row - 1, col, index + 1) ||
                 searchWord(grid, word, row, col + 1, index + 1) ||
                 searchWord(grid, word, row, col - 1, index + 1);

    grid[row][col] = temp;
    return found;
}

bool exist(vector<vector<char>>& grid, string word) {
    for (int i = 0; i < grid.size(); i++) {
        for (int j = 0; j < grid[0].size(); j++) {
            if (searchWord(grid, word, i, j, 0))
                return true;
        }
    }
    return false;
}

int main() {
    vector<vector<char>> grid = {
        {'A', 'B', 'C', 'E'},
        {'S', 'F', 'C', 'S'},
        {'A', 'D', 'E', 'E'}
    };

    string word = "ABCCED";
    cout << "Word '" << word << "' found: " << (exist(grid, word) ? "Yes" : "No") << endl;

    return 0;
}

function searchWord(grid, word, row, col, index) {
    if (index === word.length) return true;

    const m = grid.length;
    const n = grid[0].length;

    if (row < 0 || row >= m || col < 0 || col >= n || grid[row][col] !== word[index])
        return false;

    const temp = grid[row][col];
    grid[row][col] = '#';

    const found = searchWord(grid, word, row + 1, col, index + 1) ||
                  searchWord(grid, word, row - 1, col, index + 1) ||
                  searchWord(grid, word, row, col + 1, index + 1) ||
                  searchWord(grid, word, row, col - 1, index + 1);

    grid[row][col] = temp;
    return found;
}

function exist(grid, word) {
    for (let i = 0; i < grid.length; i++) {
        for (let j = 0; j < grid[0].length; j++) {
            if (searchWord(grid, word, i, j, 0))
                return true;
        }
    }
    return false;
}

// Example
const grid = [
    ['A', 'B', 'C', 'E'],
    ['S', 'F', 'C', 'S'],
    ['A', 'D', 'E', 'E']
];

const word = "ABCCED";
console.log(`Word '${word}' found: ${exist(grid, word) ? 'Yes' : 'No'}`);

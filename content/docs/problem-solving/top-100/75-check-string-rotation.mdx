## <TranslatedText en="Approach: Concatenation Trick" bn="পদ্ধতি: সংযুক্তি ট্রিক" />

<Tabs items={['C++', 'Python', 'JavaScript']} groupId="language">

```cpp
#include <iostream>
using namespace std;

bool isRotation(const string& s1, const string& s2) {
    if (s1.length() != s2.length()) {
        return false;
    }

    // s2 is rotation of s1 if s2 is substring of s1+s1
    string concat = s1 + s1;
    return concat.find(s2) != string::npos;
}

int main() {
    cout << boolalpha;
    cout << isRotation("waterbottle", "erbottlewat") << endl;
    cout << isRotation("hello", "lohel") << endl;
    cout << isRotation("hello", "world") << endl;
    return 0;
}
```

```python
def is_rotation(s1, s2):
    if len(s1) != len(s2):
        return False

    # s2 is rotation if it's in s1+s1
    return s2 in (s1 + s1)

print(is_rotation("waterbottle", "erbottlewat"))
print(is_rotation("hello", "lohel"))
print(is_rotation("hello", "world"))
```

```javascript
function isRotation(s1, s2) {
    if (s1.length !== s2.length) {
        return false;
    }

    return (s1 + s1).includes(s2);
}

console.log(isRotation("waterbottle", "erbottlewat"));
console.log(isRotation("hello", "lohel"));
console.log(isRotation("hello", "world"));
```

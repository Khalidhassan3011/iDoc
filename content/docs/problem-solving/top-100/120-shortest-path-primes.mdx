```python
from collections import deque

def is_prime(n):
    if n < 2:
        return False
    for i in range(2, int(n**0.5) + 1):
        if n % i == 0:
            return False
    return True

def shortest_path_primes(start, end):
    if start == end:
        return 0

    visited = {start}
    queue = deque([(start, 0)])

    while queue:
        num, dist = queue.popleft()

        for i in range(4):
            for d in '0123456789':
                if i == 0 and d == '0':
                    continue

                num_str = str(num)
                new_num_str = num_str[:i] + d + num_str[i+1:]
                new_num = int(new_num_str)

                if new_num == end:
                    return dist + 1

                if new_num not in visited and is_prime(new_num):
                    visited.add(new_num)
                    queue.append((new_num, dist + 1))

    return -1

# Example
print("Shortest path from 1033 to 8179:", shortest_path_primes(1033, 8179))
```

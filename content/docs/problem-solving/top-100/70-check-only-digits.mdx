## <TranslatedText en="Problem Statement" bn="সমস্যার বিবৃতি" />

<TranslatedText
  en="Given a string, check if it contains only numeric digits. Return true if all characters are digits (0-9), false otherwise."
  bn="একটি স্ট্রিং দেওয়া হলে, এটি শুধুমাত্র সংখ্যাসূচক ডিজিট ধারণ করে কিনা তা চেক করুন। যদি সমস্ত অক্ষর ডিজিট (0-9) হয় তবে true ফেরত দিন, অন্যথায় false।"
/>

### <TranslatedText en="Example" bn="উদাহরণ" />

```
Input: "12345"
Output: true

Input: "123a45"
Output: false

Input: "123.45"
Output: false

Input: ""
Output: false

Input: "0"
Output: true
```

## <TranslatedText en="Approach 1: Character Validation" bn="পদ্ধতি ১: অক্ষর যাচাইকরণ" />

<Tabs items={['C++', 'Python', 'JavaScript']} groupId="language">
  <Tab value="C++">
```cpp
#include <iostream>
#include <string>
#include <cctype>
using namespace std;

bool isAllDigits(const string& str) {
    // Empty string is not all digits
    if (str.empty()) {
        return false;
    }

    // Check each character
    for (char ch : str) {
        if (!isdigit(ch)) {
            return false;
        }
    }

    return true;
}

// Alternative using all_of
#include <algorithm>

bool isAllDigits2(const string& str) {
    if (str.empty()) {
        return false;
    }

    return all_of(str.begin(), str.end(), ::isdigit);
}

int main() {
    cout << boolalpha;
    cout << "\"12345\": " << isAllDigits("12345") << endl;
    cout << "\"123a45\": " << isAllDigits("123a45") << endl;
    cout << "\"123.45\": " << isAllDigits("123.45") << endl;
    cout << "\"\": " << isAllDigits("") << endl;
    cout << "\"0\": " << isAllDigits("0") << endl;
    cout << "\"-123\": " << isAllDigits("-123") << endl;

    return 0;
}
```

```python
def is_all_digits(s):
    # Empty string is not all digits
    if not s:
        return False

    # Check each character
    for ch in s:
        if not ch.isdigit():
            return False

    return True

# Using built-in method
def is_all_digits2(s):
    return bool(s) and s.isdigit()

# Using all() with generator
def is_all_digits3(s):
    return bool(s) and all(ch.isdigit() for ch in s)
```

```javascript
function isAllDigits(str) {
    // Empty string is not all digits
    if (!str) {
        return false;
    }

    // Check each character
    for (const ch of str) {
        if (ch < '0' || ch > '9') {
            return false;
        }
    }

    return true;
}

// Using every method
function isAllDigits2(str) {
    return Boolean(str) && [...str].every(ch => ch >= '0' && ch <= '9');
}

// Test cases
console.log(`"12345": ${isAllDigits("12345")}`);
console.log(`"123a45": ${isAllDigits("123a45")}`);
console.log(`"123.45": ${isAllDigits("123.45")}`);
console.log(`"": ${isAllDigits("")}`);
console.log(`"0": ${isAllDigits("0")}`);
console.log(`"-123": ${isAllDigits("-123")}`);
```

## <TranslatedText en="Approach 2: Regular Expression" bn="পদ্ধতি ২: রেগুলার এক্সপ্রেশন" />

<Tabs items={['C++', 'Python', 'JavaScript']} groupId="language">

```cpp
#include <regex>

bool isAllDigitsRegex(const string& str) {
    if (str.empty()) {
        return false;
    }

    // Pattern: one or more digits
    regex pattern("^[0-9]+$");
    return regex_match(str, pattern);
}
```
  </Tab>

```python
import re

def is_all_digits_regex(s):
    if not s:
        return False

    # Pattern: one or more digits
    return bool(re.match(r'^\d+$', s))

# Alternative
def is_all_digits_regex2(s):
    return bool(s) and bool(re.fullmatch(r'\d+', s))
```

```javascript
function isAllDigitsRegex(str) {
    if (!str) {
        return false;
    }

    // Pattern: one or more digits
    return /^\d+$/.test(str);
}

// Alternative
function isAllDigitsRegex2(str) {
    return Boolean(str) && /^[0-9]+$/.test(str);
}
```

## <TranslatedText en="Approach" bn="পদ্ধতি" />

```cpp
#include <iostream>
#include <algorithm>
using namespace std;

string removeCharacter(string s, char ch) {
    string result;
    for (char c : s) {
        if (c != ch) {
            result += c;
        }
    }
    return result;
}

// Using erase-remove idiom
string removeCharacterSTL(string s, char ch) {
    s.erase(remove(s.begin(), s.end(), ch), s.end());
    return s;
}

int main() {
    cout << removeCharacter("hello world", 'l') << endl;
    cout << removeCharacter("programming", 'g') << endl;
    return 0;
}
```

  <Tab value="Python">
```python
def remove_character(s, ch):
    return ''.join(c for c in s if c != ch)

# Using replace
def remove_character2(s, ch):
    return s.replace(ch, '')

# Using filter
def remove_character3(s, ch):
    return ''.join(filter(lambda c: c != ch, s))

print(remove_character("hello world", 'l'))
print(remove_character("programming", 'g'))
```

  <Tab value="JavaScript">
```javascript
function removeCharacter(s, ch) {
    return s.split('').filter(c => c !== ch).join('');
}

// Using replace with regex
function removeCharacter2(s, ch) {
    const escaped = ch.replace(/[.*+?^${}()|[\]\\]/g, '\\$&');
    return s.replace(new RegExp(escaped, 'g'), '');
}

// Using replaceAll (ES2021)
function removeCharacter3(s, ch) {
    return s.replaceAll(ch, '');
}

console.log(removeCharacter("hello world", 'l'));
console.log(removeCharacter("programming", 'g'));
```

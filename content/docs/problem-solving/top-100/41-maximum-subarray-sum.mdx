### <TranslatedText en="Approach 1: Kadane's Algorithm (Optimal)" bn="পদ্ধতি ১: কাদানের অ্যালগরিদম (সর্বোত্তম)" />

<TranslatedText
  en="The most efficient approach using dynamic programming. At each position, we decide whether to extend the current subarray or start fresh from the current element."
  bn="ডায়নামিক প্রোগ্রামিং ব্যবহার করে সবচেয়ে দক্ষ পদ্ধতি। প্রতিটি অবস্থানে, আমরা সিদ্ধান্ত নিই যে বর্তমান সাবঅ্যারে বাড়াবো নাকি বর্তমান এলিমেন্ট থেকে নতুন শুরু করবো।"

```cpp
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int maxSubArray(vector<int>& nums) {
    if (nums.empty()) return 0;

    int maxSum = nums[0];
    int currentSum = nums[0];

    for (int i = 1; i < nums.size(); i++) {
        // Either extend the existing subarray or start a new one
        currentSum = max(currentSum + nums[i], nums[i]);

        // Update maximum sum found so far
        maxSum = max(maxSum, currentSum);
    }

    return maxSum;
}

int main() {
    vector<int> nums1 = {-2, 1, -3, 4, -1, 2, 1, -5, 4};
    cout << "Array: [-2, 1, -3, 4, -1, 2, 1, -5, 4]" << endl;
    cout << "Maximum subarray sum: " << maxSubArray(nums1) << endl;

    vector<int> nums2 = {5, 4, -1, 7, 8};
    cout << "\nArray: [5, 4, -1, 7, 8]" << endl;
    cout << "Maximum subarray sum: " << maxSubArray(nums2) << endl;

    vector<int> nums3 = {-1, -2, -3, -4};
    cout << "\nArray: [-1, -2, -3, -4]" << endl;
    cout << "Maximum subarray sum: " << maxSubArray(nums3) << endl;

    return 0;
}
```

### <TranslatedText en="Approach 2: Kadane's with Subarray Indices" bn="পদ্ধতি ২: সাবঅ্যারে ইনডেক্স সহ কাদানের" />

```cpp
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

struct SubarrayResult {
    int sum;
    int start;
    int end;
};

SubarrayResult maxSubArrayWithIndices(vector<int>& nums) {
    if (nums.empty()) return {0, -1, -1};

    int maxSum = nums[0];
    int currentSum = nums[0];
    int start = 0;
    int end = 0;
    int tempStart = 0;

    for (int i = 1; i < nums.size(); i++) {
        if (currentSum + nums[i] > nums[i]) {
            currentSum = currentSum + nums[i];
        } else {
            currentSum = nums[i];
            tempStart = i; // New subarray starts here
        }

        if (currentSum > maxSum) {
            maxSum = currentSum;
            start = tempStart;
            end = i;
        }
    }

    return {maxSum, start, end};
}

int main() {
    vector<int> nums = {-2, 1, -3, 4, -1, 2, 1, -5, 4};
    cout << "Array: [-2, 1, -3, 4, -1, 2, 1, -5, 4]" << endl;

    auto result = maxSubArrayWithIndices(nums);
    cout << "Sum: " << result.sum << endl;
    cout << "Start: " << result.start << ", End: " << result.end << endl;

    cout << "Subarray: [";
    for (int i = result.start; i <= result.end; i++) {
        cout << nums[i];
        if (i < result.end) cout << ", ";
    }
    cout << "]" << endl;

    return 0;
}
```

<TranslatedText
  en="Let's trace through **Kadane's Algorithm** with array `[-2, 1, -3, 4, -1, 2, 1, -5, 4]`:"
  bn="চলুন **কাদানের অ্যালগরিদম** অ্যারে `[-2, 1, -3, 4, -1, 2, 1, -5, 4]` দিয়ে ট্রেস করি:"
/>

```
Array: [-2, 1, -3, 4, -1, 2, 1, -5, 4]

Step 0 (Initialization):
  i=0, nums[0]=-2
  currentSum = -2
  maxSum = -2

Step 1: i=1, nums[1]=1
  currentSum = max(-2 + 1, 1) = max(-1, 1) = 1
  maxSum = max(-2, 1) = 1
  Decision: Start fresh from 1

Step 2: i=2, nums[2]=-3
  currentSum = max(1 + (-3), -3) = max(-2, -3) = -2
  maxSum = max(1, -2) = 1
  Decision: Extend (even though negative)

Step 3: i=3, nums[3]=4
  currentSum = max(-2 + 4, 4) = max(2, 4) = 4
  maxSum = max(1, 4) = 4
  Decision: Start fresh from 4

Step 4: i=4, nums[4]=-1
  currentSum = max(4 + (-1), -1) = max(3, -1) = 3
  maxSum = max(4, 3) = 4
  Decision: Extend to [4, -1]

Step 5: i=5, nums[5]=2
  currentSum = max(3 + 2, 2) = max(5, 2) = 5
  maxSum = max(4, 5) = 5
  Decision: Extend to [4, -1, 2]

Step 6: i=6, nums[6]=1
  currentSum = max(5 + 1, 1) = max(6, 1) = 6
  maxSum = max(5, 6) = 6 ✓
  Decision: Extend to [4, -1, 2, 1]

Step 7: i=7, nums[7]=-5
  currentSum = max(6 + (-5), -5) = max(1, -5) = 1
  maxSum = max(6, 1) = 6
  Decision: Extend (but sum decreases)

Step 8: i=8, nums[8]=4
  currentSum = max(1 + 4, 4) = max(5, 4) = 5
  maxSum = max(6, 5) = 6

Final Result: 6
Maximum Subarray: [4, -1, 2, 1]
```

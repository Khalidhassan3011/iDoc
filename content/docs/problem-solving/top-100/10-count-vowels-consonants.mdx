<Tabs items={['C++', 'Python', 'JavaScript']}>
<Tab value="C++">
```cpp
#include <iostream>
#include <string>
#include <cctype>
using namespace std;

bool isVowel(char ch) {
    ch = tolower(ch);
    return ch == 'a' || ch == 'e' || ch == 'i' || ch == 'o' || ch == 'u';
}

void countVowelsConsonants(const string& str) {
    int vowels = 0;
    int consonants = 0;

    for (char ch : str) {
        if (isalpha(ch)) {  // Check if it's an alphabet character
            if (isVowel(ch)) {
                vowels++;
            } else {
                consonants++;
            }
        }
    }

    cout << "String: \"" << str << "\"" << endl;
    cout << "Vowels: " << vowels << endl;
    cout << "Consonants: " << consonants << endl;
}

int main() {
    string text = "Hello World";

    countVowelsConsonants(text);

    // Test with more examples
    cout << "\nOther examples:" << endl;
    countVowelsConsonants("Programming123");
    countVowelsConsonants("AEIOUaeiou");

    return 0;
}
```

## <TranslatedText en="Approach 2: Using Set for Vowel Lookup" bn="পদ্ধতি ২: স্বরবর্ণ খোঁজার জন্য সেট ব্যবহার করে" />

```cpp
#include <iostream>
#include <string>
#include <unordered_set>
#include <cctype>
using namespace std;

void countVowelsConsonantsSet(const string& str) {
    unordered_set<char> vowels = {'a', 'e', 'i', 'o', 'u',
                                   'A', 'E', 'I', 'O', 'U'};

    int vowelCount = 0;
    int consonantCount = 0;

    for (char ch : str) {
        if (isalpha(ch)) {
            if (vowels.find(ch) != vowels.end()) {
                vowelCount++;
            } else {
                consonantCount++;
            }
        }
    }

    cout << "String: \"" << str << "\"" << endl;
    cout << "Vowels: " << vowelCount << endl;
    cout << "Consonants: " << consonantCount << endl;
}

int main() {
    string text = "Hello World";

    countVowelsConsonantsSet(text);

    return 0;
}
```

## <TranslatedText en="Approach 3: Detailed Count with Characters" bn="পদ্ধতি ৩: অক্ষর সহ বিস্তারিত গণনা" />

```cpp
#include <iostream>
#include <string>
#include <vector>
#include <cctype>
using namespace std;

struct CharacterCount {
    int vowelCount;
    int consonantCount;
    vector<char> vowels;
    vector<char> consonants;
};

bool isVowel(char ch) {
    ch = tolower(ch);
    return ch == 'a' || ch == 'e' || ch == 'i' || ch == 'o' || ch == 'u';
}

CharacterCount countVowelsConsonantsDetailed(const string& str) {
    CharacterCount result = {0, 0, {}, {}};

    for (char ch : str) {
        if (isalpha(ch)) {
            if (isVowel(ch)) {
                result.vowelCount++;
                result.vowels.push_back(ch);
            } else {
                result.consonantCount++;
                result.consonants.push_back(ch);
            }
        }
    }

    return result;
}

void printCharacterCount(const string& str) {
    CharacterCount result = countVowelsConsonantsDetailed(str);

    cout << "String: \"" << str << "\"" << endl;
    cout << "Vowels: " << result.vowelCount << " (";
    for (int i = 0; i < result.vowels.size(); i++) {
        cout << result.vowels[i];
        if (i < result.vowels.size() - 1) cout << ", ";
    }
    cout << ")" << endl;

    cout << "Consonants: " << result.consonantCount << " (";
    for (int i = 0; i < result.consonants.size(); i++) {
        cout << result.consonants[i];
        if (i < result.consonants.size() - 1) cout << ", ";
    }
    cout << ")" << endl;
}

int main() {
    string text = "Hello World";

    printCharacterCount(text);

    return 0;
}
```

## <TranslatedText en="Approach 4: Using Regular Expressions" bn="পদ্ধতি ৪: রেগুলার এক্সপ্রেশন ব্যবহার করে" />

```cpp
#include <iostream>
#include <string>
#include <regex>
using namespace std;

void countVowelsConsonantsRegex(const string& str) {
    // Count vowels using regex
    regex vowelPattern("[aeiouAEIOU]");
    auto vowelsBegin = sregex_iterator(str.begin(), str.end(), vowelPattern);
    auto vowelsEnd = sregex_iterator();
    int vowelCount = distance(vowelsBegin, vowelsEnd);

    // Count consonants using regex
    regex consonantPattern("[bcdfghjklmnpqrstvwxyzBCDFGHJKLMNPQRSTVWXYZ]");
    auto consonantsBegin = sregex_iterator(str.begin(), str.end(), consonantPattern);
    auto consonantsEnd = sregex_iterator();
    int consonantCount = distance(consonantsBegin, consonantsEnd);

    cout << "String: \"" << str << "\"" << endl;
    cout << "Vowels: " << vowelCount << endl;
    cout << "Consonants: " << consonantCount << endl;
}

int main() {
    string text = "Hello World";

    countVowelsConsonantsRegex(text);

    return 0;
}
```

```python
import re

def count_vowels_consonants_regex(s):
    # Count vowels
    vowels = re.findall(r'[aeiouAEIOU]', s)
    vowel_count = len(vowels)

    # Count consonants
    consonants = re.findall(r'[bcdfghjklmnpqrstvwxyzBCDFGHJKLMNPQRSTVWXYZ]', s)
    consonant_count = len(consonants)

    return vowel_count, consonant_count
```

```javascript
function countVowelsConsonantsRegex(str) {
    // Count vowels
    const vowelMatches = str.match(/[aeiouAEIOU]/g);
    const vowelCount = vowelMatches ? vowelMatches.length : 0;

    // Count consonants
    const consonantMatches = str.match(/[bcdfghjklmnpqrstvwxyzBCDFGHJKLMNPQRSTVWXYZ]/g);
    const consonantCount = consonantMatches ? consonantMatches.length : 0;
```

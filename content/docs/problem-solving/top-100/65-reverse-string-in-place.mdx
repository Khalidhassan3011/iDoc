```cpp
#include <iostream>
#include <algorithm>
using namespace std;

void reverseString(string& s) {
    int left = 0;
    int right = s.length() - 1;

    while (left < right) {
        // Swap characters
        swap(s[left], s[right]);
        left++;
        right--;
    }
}

// Alternative using standard library
void reverseStringSTL(string& s) {
    reverse(s.begin(), s.end());
}

int main() {
    string s1 = "hello";
    cout << "Original: " << s1 << endl;
    reverseString(s1);
    cout << "Reversed: " << s1 << endl;

    string s2 = "Java";
    cout << "\nOriginal: " << s2 << endl;
    reverseStringSTL(s2);
    cout << "Reversed: " << s2 << endl;

    string s3 = "a";
    cout << "\nOriginal: " << s3 << endl;
    reverseString(s3);
    cout << "Reversed: " << s3 << endl;

    return 0;
}
```

**Output:**

def reverse_string(s):
    # Convert to list (strings are immutable in Python)
    chars = list(s)
    left = 0
    right = len(chars) - 1

    while left < right:
        # Swap characters
        chars[left], chars[right] = chars[right], chars[left]
        left += 1
        right -= 1

    return ''.join(chars)

# Pythonic way (not truly in-place)
def reverse_string_pythonic(s):
    return s[::-1]

# Test cases
s1 = "hello"
print(f"Original: {s1}")
print(f"Reversed: {reverse_string(s1)}")

print(f"Reversed: {reverse_string_pythonic(s2)}")

```javascript
function reverseString(s) {
    // Convert to array (strings are immutable in JavaScript)
    const chars = s.split('');
    let left = 0;
    let right = chars.length - 1;

    while (left < right) {
        // Swap characters
        [chars[left], chars[right]] = [chars[right], chars[left]];
        left++;
        right--;
    }

    return chars.join('');
}

// Alternative methods
function reverseStringBuiltIn(s) {
    return s.split('').reverse().join('');
}

function reverseStringRecursive(s) {
    if (s.length <= 1) return s;
    return reverseStringRecursive(s.slice(1)) + s[0];
}

// Test cases
const s1 = "hello";
console.log(`Original: ${s1}`);
console.log(`Reversed: ${reverseString(s1)}`);

const s2 = "Java";
console.log(`\nOriginal: ${s2}`);
console.log(`Reversed: ${reverseStringBuiltIn(s2)}`);

const s3 = "a";
console.log(`\nOriginal: ${s3}`);
console.log(`Reversed: ${reverseString(s3)}`);
```

**Output:**

Reversed: avaJ

Reversed: a

## <TranslatedText en="How It Works" bn="এটি কীভাবে কাজ করে" />

<TranslatedText
  en="The two-pointer approach works by swapping characters from both ends moving toward the center:"
  bn="টু-পয়েন্টার পদ্ধতি কেন্দ্রের দিকে চলমান উভয় প্রান্ত থেকে অক্ষর অদলবদল করে কাজ করে:"
/>

```
Example: Reverse "hello"

Initial: h e l l o
         ↑       ↑
         L       R

Step 1: Swap h and o
        o e l l h
          ↑   ↑
          L   R

Step 2: Swap e and l
        o l l e h
            ↑
           L,R (meet in middle)

Result: "olleh"
```

## <TranslatedText en="Common Mistakes to Avoid" bn="এড়ানোর জন্য সাধারণ ভুল" />

<TranslatedText
  en={
    <>
      <ul>
        <li>Not checking for null or empty strings</li>
        <li>Forgetting that strings are immutable in most languages</li>
        <li>Using O(n) extra space when in-place is possible</li>
        <li>Off-by-one errors in pointer initialization or loop condition</li>
        <li>Not handling single character strings</li>
        <li>Swapping when pointers have crossed (should be left less than right)</li>
      </ul>

  }

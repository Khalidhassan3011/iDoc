### <TranslatedText en="Approach 1: Using Hash Set" bn="পদ্ধতি ১: হ্যাশ সেট ব্যবহার করে" />

<TranslatedText
  en={<p>Use a hash set to track seen numbers and find the duplicate. Then find missing number.</p>}
  bn={<p>দেখা সংখ্যা ট্র্যাক করতে এবং ডুপ্লিকেট খুঁজতে হ্যাশ সেট ব্যবহার করুন। তারপর অনুপস্থিত সংখ্যা খুঁজুন।</p>}

<Tabs items={['C++', 'Python', 'JavaScript']} groupId="language">
  <Tab value="C++">
```cpp
#include <iostream>
#include <vector>
#include <unordered_set>
using namespace std;

pair<int, int> findMissingRepeating(vector<int>& arr) {
    int n = arr.size();
    unordered_set<int> seen;
    int repeating = -1;

    // Find repeating number
    for (int num : arr) {
        if (seen.count(num)) {
            repeating = num;
        }
        seen.insert(num);
    }

    // Find missing number
    int missing = -1;
    for (int i = 1; i <= n; i++) {
        if (!seen.count(i)) {
            missing = i;
            break;
        }
    }

    return {repeating, missing};
}

int main() {
    vector<int> arr = {3, 1, 2, 5, 3};

    auto result = findMissingRepeating(arr);
    cout << "Repeating: " << result.first << endl;
    cout << "Missing: " << result.second << endl;

    return 0;
}
```

**Output:**
```
Repeating: 3
Missing: 4
```

```python
def find_missing_repeating(arr):
    n = len(arr)
    seen = set()
    repeating = -1

    # Find repeating number
    for num in arr:
        if num in seen:
            repeating = num
        seen.add(num)

    # Find missing number
    missing = -1
    for i in range(1, n + 1):
        if i not in seen:
            missing = i
            break

    return repeating, missing
```

### <TranslatedText en="Approach 2: Using Sum Formula" bn="পদ্ধতি ২: যোগফল সূত্র ব্যবহার করে" />

<TranslatedText
  en={<p>Use mathematical formulas: sum of 1 to n and sum of squares to find both numbers.</p>}
  bn={<p>গাণিতিক সূত্র ব্যবহার করুন: উভয় সংখ্যা খুঁজতে 1 থেকে n পর্যন্ত যোগফল এবং বর্গের যোগফল।</p>}
/>

<Tabs items={['C++', 'Python', 'JavaScript']} groupId="language">
  <Tab value="C++">
```cpp
#include <iostream>
#include <vector>
#include <cmath>
using namespace std;

pair<int, int> findMissingRepeatingMath(vector<int>& arr) {
    long long n = arr.size();

    // Calculate sum and sum of squares
    long long actualSum = 0, actualSumSq = 0;
    for (int num : arr) {
        actualSum += num;
        actualSumSq += (long long)num * num;
    }

    // Expected sum: n * (n + 1) / 2
    long long expectedSum = n * (n + 1) / 2;

    // Expected sum of squares: n * (n + 1) * (2n + 1) / 6
    long long expectedSumSq = n * (n + 1) * (2 * n + 1) / 6;

    // Let repeating = x, missing = y
    // x - y = actualSum - expectedSum
    long long diff = actualSum - expectedSum;

    // x^2 - y^2 = actualSumSq - expectedSumSq
    // (x + y)(x - y) = actualSumSq - expectedSumSq
    long long sumSqDiff = actualSumSq - expectedSumSq;

    // x + y = sumSqDiff / diff
    long long sum = sumSqDiff / diff;

    // Solve: x - y = diff, x + y = sum
    int repeating = (diff + sum) / 2;
    int missing = (sum - diff) / 2;

    return {repeating, missing};
}

int main() {
    vector<int> arr = {3, 1, 2, 5, 3};

    auto result = findMissingRepeatingMath(arr);
    cout << "Repeating: " << result.first << endl;
    cout << "Missing: " << result.second << endl;

    return 0;
}
```

**Output:**
```
Repeating: 3
```

```cpp
#include <iostream>
#include <vector>
using namespace std;

pair<int, int> findMissingRepeatingXOR(vector<int>& arr) {
    int n = arr.size();

    // XOR all array elements and numbers 1 to n
    // This gives us: repeating XOR missing
    int xorAll = 0;
    for (int num : arr) {
        xorAll ^= num;
    }
    for (int i = 1; i <= n; i++) {
        xorAll ^= i;
    }

    // Find rightmost set bit
    int setBit = xorAll & ~(xorAll - 1);

    // Divide numbers into two groups based on set bit
    int x = 0, y = 0;

    for (int num : arr) {
        if (num & setBit) {
            x ^= num;
        } else {
            y ^= num;
        }
    }

    for (int i = 1; i <= n; i++) {
        if (i & setBit) {
            x ^= i;
        } else {
            y ^= i;
        }
    }

    // Determine which is repeating and which is missing
    int repeating = -1, missing = -1;
    for (int num : arr) {
        if (num == x) {
            repeating = x;
            missing = y;
            break;
        } else if (num == y) {
            repeating = y;
            missing = x;
            break;
        }
    }

    return {repeating, missing};
}

int main() {
    vector<int> arr = {3, 1, 2, 5, 3};

    auto result = findMissingRepeatingXOR(arr);
    cout << "Repeating: " << result.first << endl;
    cout << "Missing: " << result.second << endl;

    return 0;
}
```

### <TranslatedText en="Example" bn="উদাহরণ" />


## <TranslatedText en="Approach 1: Iterative Count" bn="পদ্ধতি ১: পুনরাবৃত্তিমূলক গণনা" />

```cpp
#include <iostream>
#include <string>
#include <algorithm>
#include <cctype>
using namespace std;

int countOccurrence(const string& str, char ch) {
    int count = 0;

    for (char c : str) {
        if (c == ch) {
            count++;
        }
    }

    return count;
}

// Case-insensitive version
int countOccurrenceIgnoreCase(const string& str, char ch) {
    int count = 0;
    char lowerCh = tolower(ch);

    for (char c : str) {
        if (tolower(c) == lowerCh) {
            count++;
        }
    }

    return count;
}

// Using STL count
int countOccurrenceSTL(const string& str, char ch) {
    return count(str.begin(), str.end(), ch);
}

int main() {
    string s1 = "hello world";
    cout << "Count of 'l' in \"" << s1 << "\": "
         << countOccurrence(s1, 'l') << endl;

    string s2 = "programming";
    cout << "Count of 'g' in \"" << s2 << "\": "
         << countOccurrence(s2, 'g') << endl;

    string s3 = "Java";
    cout << "Count of 'a' in \"" << s3 << "\" (case-sensitive): "
         << countOccurrence(s3, 'a') << endl;
    cout << "Count of 'A' in \"" << s3 << "\" (case-insensitive): "
         << countOccurrenceIgnoreCase(s3, 'A') << endl;

    string s4 = "hello";
    cout << "Count of 'x' in \"" << s4 << "\": "
         << countOccurrence(s4, 'x') << endl;

    return 0;
}
```

```python
def count_occurrence(s, ch):
    count = 0

    for c in s:
        if c == ch:
            count += 1

    return count

# Case-insensitive version
def count_occurrence_ignore_case(s, ch):
    count = 0
    ch_lower = ch.lower()

    for c in s:
        if c.lower() == ch_lower:
            count += 1

    return count

# Using built-in count method
def count_occurrence_builtin(s, ch):
    return s.count(ch)

# Using sum with generator
def count_occurrence_sum(s, ch):
    return sum(1 for c in s if c == ch)
```

```javascript
function countOccurrence(str, ch) {
    let count = 0;

    for (const c of str) {
        if (c === ch) {
            count++;
        }
    }

    return count;
}

// Case-insensitive version
function countOccurrenceIgnoreCase(str, ch) {
    let count = 0;
    const chLower = ch.toLowerCase();

    for (const c of str) {
        if (c.toLowerCase() === chLower) {
            count++;
        }
    }

    return count;
}

// Using split method
function countOccurrenceSplit(str, ch) {
    return str.split(ch).length - 1;
}

// Using filter
function countOccurrenceFilter(str, ch) {
    return [...str].filter(c => c === ch).length;
}

// Using reduce
function countOccurrenceReduce(str, ch) {
    return [...str].reduce((count, c) => count + (c === ch ? 1 : 0), 0);
}

// Test cases
const s1 = "hello world";
console.log(`Count of 'l' in "${s1}": ${countOccurrence(s1, 'l')}`);

const s2 = "programming";
console.log(`Count of 'g' in "${s2}": ${countOccurrence(s2, 'g')}`);

const s3 = "Java";
console.log(`Count of 'a' in "${s3}" (case-sensitive): ${countOccurrence(s3, 'a')}`);
console.log(`Count of 'A' in "${s3}" (case-insensitive): ${countOccurrenceIgnoreCase(s3, 'A')}`);

const s4 = "hello";
console.log(`Count of 'x' in "${s4}": ${countOccurrence(s4, 'x')}`);
```

```cpp
#include <regex>

int countOccurrenceRegex(const string& str, char ch) {
    string pattern(1, ch);
    regex re(pattern);

    auto begin = sregex_iterator(str.begin(), str.end(), re);
    auto end = sregex_iterator();

    return distance(begin, end);
}
```

```python
import re

def count_occurrence_regex(s, ch):
    return len(re.findall(re.escape(ch), s))

# Case-insensitive with regex
def count_occurrence_regex_ignore_case(s, ch):
    return len(re.findall(re.escape(ch), s, re.IGNORECASE))
```

```javascript
function countOccurrenceRegex(str, ch) {
    // Escape special regex characters
    const escaped = ch.replace(/[.*+?^${}()|[\]\\]/g, '\\$&');
    const regex = new RegExp(escaped, 'g');
    const matches = str.match(regex);
    return matches ? matches.length : 0;
}

// Case-insensitive
function countOccurrenceRegexIgnoreCase(str, ch) {
    const escaped = ch.replace(/[.*+?^${}()|[\]\\]/g, '\\$&');
    const regex = new RegExp(escaped, 'gi');
    const matches = str.match(regex);
    return matches ? matches.length : 0;
}
```

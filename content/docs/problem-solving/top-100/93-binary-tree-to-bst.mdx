#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

struct TreeNode {
    int val;
    TreeNode* left;
    TreeNode* right;
    TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
};

void extractInorder(TreeNode* root, vector<int>& values) {
    if (!root) return;
    extractInorder(root->left, values);
    values.push_back(root->val);
    extractInorder(root->right, values);
}

void fillInorder(TreeNode* root, vector<int>& values, int& index) {
    if (!root) return;
    fillInorder(root->left, values, index);
    root->val = values[index++];
    fillInorder(root->right, values, index);
}

void binaryTreeToBST(TreeNode* root) {
    vector<int> values;
    extractInorder(root, values);
    sort(values.begin(), values.end());
    int index = 0;
    fillInorder(root, values, index);
}

void printInorder(TreeNode* root) {
    if (!root) return;
    printInorder(root->left);
    cout << root->val << " ";
    printInorder(root->right);
}

int main() {
    TreeNode* root = new TreeNode(10);
    root->left = new TreeNode(30);
    root->right = new TreeNode(15);
    root->left->left = new TreeNode(20);
    root->right->right = new TreeNode(5);

    cout << "Before conversion: ";
    printInorder(root);
    cout << endl;

    binaryTreeToBST(root);

    cout << "After conversion: ";
    printInorder(root);
    cout << endl;

    return 0;
}
```

**Output:**
```
Before conversion: 20 30 10 15 5
After conversion: 5 10 15 20 30
```

class TreeNode {
    constructor(val, left = null, right = null) {
        this.val = val;
        this.left = left;
        this.right = right;
    }
}

function extractInorder(root, values) {
    if (!root) return;
    extractInorder(root.left, values);
    values.push(root.val);
    extractInorder(root.right, values);
}

function fillInorder(root, values, index) {
    if (!root) return index;
    index = fillInorder(root.left, values, index);
    root.val = values[index.value];
    index.value++;
    index = fillInorder(root.right, values, index);
    return index;
}

function binaryTreeToBST(root) {
    const values = [];
    extractInorder(root, values);
    values.sort((a, b) => a - b);
    fillInorder(root, values, { value: 0 });
}

function printInorder(root) {
    if (!root) return;
    printInorder(root.left);
    process.stdout.write(root.val + " ");
    printInorder(root.right);
}

const root = new TreeNode(10);
root.left = new TreeNode(30);
root.right = new TreeNode(15);
root.left.left = new TreeNode(20);
root.right.right = new TreeNode(5);

process.stdout.write("Before conversion: ");
printInorder(root);
console.log();

binaryTreeToBST(root);

process.stdout.write("After conversion: ");
printInorder(root);
console.log();
```

```cpp
#include <iostream>
using namespace std;

void addMatrix2x2(int a[2][2], int b[2][2], int result[2][2]) {
    for (int i = 0; i < 2; i++) {
        for (int j = 0; j < 2; j++) {
            result[i][j] = a[i][j] + b[i][j];
        }
    }
}

void printMatrix2x2(int matrix[2][2]) {
    for (int i = 0; i < 2; i++) {
        cout << "[";
        for (int j = 0; j < 2; j++) {
            cout << matrix[i][j];
            if (j < 1) cout << "  ";
        }
        cout << "]" << endl;
    }
}

int main() {
    int matrixA[2][2] = {{1, 2}, {3, 4}};
    int matrixB[2][2] = {{5, 6}, {7, 8}};
    int result[2][2];

    addMatrix2x2(matrixA, matrixB, result);

    cout << "Matrix A:" << endl;
    printMatrix2x2(matrixA);

    cout << "\nMatrix B:" << endl;
    printMatrix2x2(matrixB);

    cout << "\nResult (A + B):" << endl;
    printMatrix2x2(result);

    return 0;
}
```

## <TranslatedText en="Approach 3: Generic Matrix Addition (Any Size)" bn="পদ্ধতি ৩: জেনেরিক ম্যাট্রিক্স যোগ (যেকোনো আকার)" />

```cpp
#include <iostream>
#include <vector>
using namespace std;

// Generic function for any matrix size
vector<vector<int>> addMatrix(const vector<vector<int>>& a,
                               const vector<vector<int>>& b) {
    int rows = a.size();
    int cols = a[0].size();

    // Check if dimensions match
    if (rows != b.size() || cols != b[0].size()) {
        cout << "Error: Matrices must have same dimensions!" << endl;
        return {};
    }

    vector<vector<int>> result(rows, vector<int>(cols));

    for (int i = 0; i < rows; i++) {
        for (int j = 0; j < cols; j++) {
            result[i][j] = a[i][j] + b[i][j];
        }
    }

    return result;
}

void printMatrix(const vector<vector<int>>& matrix) {
    for (const auto& row : matrix) {
        cout << "[";
        for (int i = 0; i < row.size(); i++) {
            cout << row[i];
            if (i < row.size() - 1) cout << "  ";
        }
        cout << "]" << endl;
    }
}

int main() {
    // Test with 2x2
    vector<vector<int>> matrixA2 = {{1, 2}, {3, 4}};
    vector<vector<int>> matrixB2 = {{5, 6}, {7, 8}};

    cout << "2x2 Matrix Addition:" << endl;
    cout << "Matrix A:" << endl;
    printMatrix(matrixA2);
    cout << "\nMatrix B:" << endl;
    printMatrix(matrixB2);
    cout << "\nResult:" << endl;
    printMatrix(addMatrix(matrixA2, matrixB2));

    // Test with 3x3
    vector<vector<int>> matrixA3 = {{1, 2, 3}, {4, 5, 6}, {7, 8, 9}};
    vector<vector<int>> matrixB3 = {{9, 8, 7}, {6, 5, 4}, {3, 2, 1}};

    cout << "\n\n3x3 Matrix Addition:" << endl;
    cout << "Matrix A:" << endl;
    printMatrix(matrixA3);
    cout << "\nMatrix B:" << endl;
    printMatrix(matrixB3);
    cout << "\nResult:" << endl;
    printMatrix(addMatrix(matrixA3, matrixB3));

    return 0;
}
```

```python
def add_matrix_generic(a, b):
    rows = len(a)
    cols = len(a[0])

    # Check if dimensions match
    if rows != len(b) or cols != len(b[0]):
        print("Error: Matrices must have same dimensions!")
        return None

    result = [[0] * cols for _ in range(rows)]

    for i in range(rows):
        for j in range(cols):
            result[i][j] = a[i][j] + b[i][j]

    return result
```

```javascript
function addMatrixGeneric(a, b) {
    const rows = a.length;
    const cols = a[0].length;

    // Check if dimensions match
    if (rows !== b.length || cols !== b[0].length) {
        console.log("Error: Matrices must have same dimensions!");
        return null;
    }

    const result = Array(rows).fill(0).map(() => Array(cols).fill(0));

    for (let i = 0; i < rows; i++) {
        for (let j = 0; j < cols; j++) {
            result[i][j] = a[i][j] + b[i][j];
        }
    }

    return result;
}

function printMatrix(matrix) {
    if (!matrix) return;
    matrix.forEach(row => {
        console.log("[" + row.join("  ") + "]");
    });
}

// Test with 2x2
const matrixA2 = [[1, 2], [3, 4]];
const matrixB2 = [[5, 6], [7, 8]];

console.log("2x2 Matrix Addition:");
console.log("Matrix A:");
printMatrix(matrixA2);
console.log("\nMatrix B:");
printMatrix(matrixB2);
```

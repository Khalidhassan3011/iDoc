### <TranslatedText en="Approach 1: Two Pointers (Optimal)" bn="পদ্ধতি ১: টু পয়েন্টার (সর্বোত্তম)" />

<TranslatedText

  bn="উভয় প্রান্ত থেকে দুটি পয়েন্টার ব্যবহার করে এলিমেন্ট সোয়াপ করুন। এটি সবচেয়ে কার্যকর এবং সাধারণভাবে ব্যবহৃত পদ্ধতি।"
/>

**<TranslatedText en="Time Complexity" bn="সময় জটিলতা" />:** O(n)
**<TranslatedText en="Space Complexity" bn="স্থান জটিলতা" />:** O(1)

<Tabs items={['C++', 'Python', 'JavaScript']} groupId="language">
  <Tab value="C++">
```cpp
#include <iostream>
#include <vector>
using namespace std;

void reverseArray(vector<int>& arr) {
    int left = 0;
    int right = arr.size() - 1;

    while (left < right) {
        // Swap elements at left and right
        int temp = arr[left];
        arr[left] = arr[right];
        arr[right] = temp;

        left++;
        right--;
    }
}

int main() {
    vector<int> arr1 = {1, 2, 3, 4, 5};
    cout << "Original: ";
    for (int x : arr1) cout << x << " ";

    reverseArray(arr1);

    cout << "\nReversed: ";
    for (int x : arr1) cout << x << " ";
    cout << endl;

    return 0;
}
```

```python
def reverse_array(arr):
    """Reverse array in place using two pointers"""
    left = 0
    right = len(arr) - 1

    while left < right:
        # Swap elements at left and right
        arr[left], arr[right] = arr[right], arr[left]

        left += 1
        right -= 1
```

```cpp
#include <iostream>
#include <vector>
using namespace std;

void reverseArrayForLoop(vector<int>& arr) {
    int n = arr.size();

    for (int i = 0; i < n / 2; i++) {
        // Swap arr[i] with arr[n-1-i]
        int temp = arr[i];
        arr[i] = arr[n - 1 - i];
        arr[n - 1 - i] = temp;
    }
}

int main() {
    vector<int> arr = {1, 2, 3, 4, 5};
    cout << "Original: ";
    for (int x : arr) cout << x << " ";

    reverseArrayForLoop(arr);

    cout << "\nReversed: ";
    for (int x : arr) cout << x << " ";
    cout << endl;

    return 0;
}
```

**<TranslatedText en="Space Complexity" bn="স্থান জটিলতা" />:** O(n) <TranslatedText en="for recursion stack" bn="রিকার্শন স্ট্যাকের জন্য" />

```cpp
#include <iostream>
#include <vector>
using namespace std;

void reverseArrayRecursive(vector<int>& arr, int left, int right) {
    if (left >= right) return;

    // Swap
    int temp = arr[left];
    arr[left] = arr[right];
    arr[right] = temp;

    // Recursive call
    reverseArrayRecursive(arr, left + 1, right - 1);
}

int main() {
    vector<int> arr = {1, 2, 3, 4, 5};
    cout << "Original: ";
    for (int x : arr) cout << x << " ";

    reverseArrayRecursive(arr, 0, arr.size() - 1);

    cout << "\nReversed: ";
    for (int x : arr) cout << x << " ";
    cout << endl;

    return 0;
}
```
